function updateClock() {
  var now = new Date();
  document.getElementById("current-date").textContent = now.toLocaleDateString("en-US", { weekday: "long", year: "numeric", month: "long", day: "numeric" });
  document.getElementById("current-time").textContent = now.toLocaleTimeString("en-US", { hour: "2-digit", minute: "2-digit", timeZoneName: "short" });
}
updateClock();
setInterval(updateClock, 60000);

function fmtNum(n, decimals) {
  if (n === null || n === undefined || isNaN(n)) return "—";
  return parseFloat(n).toLocaleString("en-US", { minimumFractionDigits: decimals || 0, maximumFractionDigits: decimals || 0 });
}
function fmtPct(n) {
  if (n === null || n === undefined || isNaN(n)) return "—";
  var v = parseFloat(n);
  return (v >= 0 ? "+" : "") + v.toFixed(2) + "%";
}
function chgClass(n) {
  var v = parseFloat(n);
  if (v > 0) return "up";
  if (v < 0) return "dn";
  return "flat";
}
function fmtMoney(n) {
  var v = parseFloat(n);
  if (v >= 1e9) return "$" + (v / 1e9).toFixed(1) + "B";
  if (v >= 1e6) return "$" + (v / 1e6).toFixed(1) + "M";
  if (v >= 1e3) return "$" + (v / 1e3).toFixed(0) + "K";
  return "$" + v.toFixed(0);
}
function timeAgo(dateStr) {
  var now = new Date();
  var then = new Date(dateStr);
  var diff = Math.floor((now - then) / 60000);
  if (diff < 60) return diff + "m ago";
  if (diff < 1440) return Math.floor(diff / 60) + "h ago";
  return Math.floor(diff / 1440) + "d ago";
}

async function fetchQuote(symbol) {
  try {
    var url = "https://query1.finance.yahoo.com/v8/finance/chart/" + encodeURIComponent(symbol) + "?interval=1d&range=1d";
    var res = await fetch(url);
    var data = await res.json();
    var meta = data.chart.result[0].meta;
    return {
      name: meta.symbol,
      price: meta.regularMarketPrice,
      prevClose: meta.chartPreviousClose || meta.previousClose,
      change: meta.regularMarketPrice - (meta.chartPreviousClose || meta.previousClose),
      changePct: ((meta.regularMarketPrice - (meta.chartPreviousClose || meta.previousClose)) / (meta.chartPreviousClose || meta.previousClose)) * 100
    };
  } catch(e) { return null; }
}

async function loadUSMarkets() {
  var container = document.getElementById("us-markets");
  var timeEl = document.getElementById("markets-time");
  timeEl.textContent = "Fetching...";

  var quotes = await Promise.all(US_SYMBOLS.map(function(s) {
    return fetchQuote(s.sym).then(function(q) { return { label: s.name, data: q }; });
  }));

  var html = "";
  quotes.forEach(function(q) {
    var d = q.data;
    var cls = d ? chgClass(d.changePct) : "flat";
    html += "<div class=\"market-card\">";
    html += "<div class=\"market-name\">" + q.label + "</div>";
    html += "<div class=\"market-value\">" + (d ? fmtNum(d.price, 2) : "—") + "</div>";
    html += "<div class=\"market-change " + cls + "\">" + (d ? fmtPct(d.changePct) : "—") + "</div>";
    html += "</div>";
  });
  container.innerHTML = html;
  timeEl.textContent = "Updated " + new Date().toLocaleTimeString("en-US", { hour: "2-digit", minute: "2-digit" });

  buildTicker(quotes);
}

async function loadIntlMarkets() {
  var container = document.getElementById("intl-markets");
  var quotes = await Promise.all(INTL_SYMBOLS.map(function(s) {
    return fetchQuote(s.sym).then(function(q) { return { label: s.name, data: q }; });
  }));

  var html = "";
  quotes.forEach(function(q) {
    var d = q.data;
    var cls = d ? chgClass(d.changePct) : "";
    html += "<div class=\"intl-item\">";
    html += "<div class=\"intl-name\">" + q.label + "</div>";
    html += "<div class=\"intl-val " + cls + "\">" + (d ? fmtNum(d.price, 2) : "—") + "</div>";
    html += "<div class=\"intl-chg " + cls + "\">" + (d ? fmtPct(d.changePct) : "—") + "</div>";
    html += "</div>";
  });
  container.innerHTML = html;
}

function buildTicker(quotes) {
  var items = quotes.map(function(q) {
    var d = q.data;
    var cls = d ? chgClass(d.changePct) : "";
    return "<span class=\"ticker-item\"><span class=\"sym\">" + q.label + "</span><span class=\"val\">" + (d ? fmtNum(d.price, 2) : "—") + "</span><span class=\"chg " + cls + "\">" + (d ? fmtPct(d.changePct) : "") + "</span></span>";
  }).join("");
  var doubled = items + items;
  document.getElementById("tickerInner").innerHTML = doubled;
}

async function fetchFRED(seriesId) {
  try {
    var url = "https://api.stlouisfed.org/fred/series/observations?series_id=" + seriesId + "&api_key=" + FRED_KEY + "&file_type=json&sort_order=desc&limit=2";
    var res = await fetch(url);
    var data = await res.json();
    var obs = data.observations;
    if (!obs || !obs.length) return null;
    return { current: obs[0], prev: obs[1] || null };
  } catch(e) { return null; }
}

async function loadMacro() {
  var grid = document.getElementById("macro-grid");
  var results = await Promise.all(FRED_SERIES.map(function(s) {
    return fetchFRED(s.id).then(function(d) { return { series: s, data: d }; });
  }));

  var html = "";
  results.forEach(function(r) {
    var s = r.series;
    var d = r.data;
    var val = d ? parseFloat(d.current.value) : null;
    var prev = d && d.prev ? parseFloat(d.prev.value) : null;
    var displayVal = "—";
    var delta = "";

    if (val !== null && !isNaN(val)) {
      if (s.scale) {
        displayVal = (val / s.scale).toFixed(1) + s.suffix;
      } else if (s.transform === "pct" && prev !== null) {
        var pctChg = ((val - prev) / prev) * 100;
        displayVal = pctChg.toFixed(1) + s.suffix;
      } else {
        displayVal = val.toFixed(2) + s.suffix;
      }
      if (prev !== null && !isNaN(prev)) {
        var diff = val - prev;
        var sign = diff >= 0 ? "+" : "";
        delta = "<div class=\"macro-delta " + (diff >= 0 ? "up" : "dn") + "\">" + sign + diff.toFixed(2) + " vs prior</div>";
      }
    }

    html += "<div class=\"macro-item\">";
    html += "<div class=\"macro-label\">" + s.label + "</div>";
    html += "<div class=\"macro-value\">" + displayVal + "</div>";
    html += "<div class=\"macro-sub\">" + s.desc + (d ? " &bull; " + d.current.date : "") + "</div>";
    html += delta;
    html += "</div>";
  });
  grid.innerHTML = html;
}

async function loadYieldCurve() {
  var container = document.getElementById("yield-curve");
  var results = await Promise.all(YIELD_SERIES.map(function(s) {
    return fetchFRED(s.id).then(function(d) { return { label: s.label, val: d ? parseFloat(d.current.value) : null }; });
  }));

  var vals = results.map(function(r) { return r.val; }).filter(function(v) { return v !== null && !isNaN(v); });
  var maxVal = vals.length ? Math.max.apply(null, vals) : 5;

  var html = "";
  results.forEach(function(r) {
    var pct = r.val !== null && !isNaN(r.val) ? Math.max(4, (r.val / maxVal) * 100) : 0;
    html += "<div class=\"yield-row\">";
    html += "<span class=\"yield-label\">" + r.label + "</span>";
    html += "<div class=\"yield-bar-wrap\"><div class=\"yield-bar\" style=\"width:" + pct + "%\"></div></div>";
    html += "<span class=\"yield-val\">" + (r.val !== null && !isNaN(r.val) ? r.val.toFixed(2) + "%" : "—") + "</span>";
    html += "</div>";
  });
  container.innerHTML = html;
}

    // Use EDGAR full text search for Form 4
    var searchRes = await fetch("https://efts.sec.gov/LATEST/search-index?forms=4&dateRange=custom&startdt=" + getDateNDaysAgo(2) + "&enddt=" + getToday());

function renderInsiderFromEDGAR(data, container) {
  var hits = data && data.hits && data.hits.hits ? data.hits.hits : [];
  if (!hits.length) { loadInsiderFallback(container); return; }

  var html = "<table class=\"data-table\"><thead><tr>";
  html += "<th>Insider</th><th>Company</th><th class=\"r\">Type</th><th class=\"r\">Filed</th><th class=\"r\">Filing</th>";
  html += "</tr></thead><tbody>";

  hits.slice(0, 20).forEach(function(h) {
    var src = h._source || {};
    var entityName = src.entity_name || src.display_names || "—";
    if (Array.isArray(entityName)) entityName = entityName.join(", ");
    var fileDate = src.file_date || "";
    var accNo = h._id || "";
    var secUrl = accNo ? "https://www.sec.gov/cgi-bin/browse-edgar?action=getcompany&filenum=&State=0&SIC=&dateb=&owner=include&count=10&search_text=&action=getcompany" : "#";

    html += "<tr>";
    html += "<td><div class=\"insider-name\">" + entityName + "</div></td>";
    html += "<td><div class=\"company-name\">Form 4 Filing</div></td>";
    html += "<td class=\"r\"><span class=\"buy-badge\">FORM 4</span></td>";
    html += "<td class=\"r\"><span class=\"date-badge\">" + (fileDate ? timeAgo(fileDate) : "—") + "</span></td>";
    html += "<td class=\"r\"><a class=\"sec-link\" href=\"https://www.sec.gov/cgi-bin/browse-edgar?action=getcompany&company=&CIK=&type=4&dateb=&owner=include&count=40&search_text=\" target=\"_blank\">SEC &#8599;</a></td>";
    html += "</tr>";
  });

  html += "</tbody></table>";
  container.innerHTML = html;
}

function loadInsiderFallback(container) {
  // Direct RSS from SEC for Form 4
  fetch("https://www.sec.gov/cgi-bin/browse-edgar?action=getcurrent&type=4&dateb=&owner=include&count=20&search_text=&output=atom")
    .then(function(r) { return r.text(); })
    .then(function(xml) {
      var parser = new DOMParser();
      var doc = parser.parseFromString(xml, "application/xml");
      var entries = doc.querySelectorAll("entry");

      var html = "<table class=\"data-table\"><thead><tr>";
      html += "<th>Filing Entity</th><th class=\"r\">Type</th><th class=\"r\">Filed</th><th class=\"r\">Link</th>";
      html += "</tr></thead><tbody>";

      var count = 0;
      entries.forEach(function(entry) {
        if (count >= 20) return;
        var title = entry.querySelector("title") ? entry.querySelector("title").textContent : "—";
        var updated = entry.querySelector("updated") ? entry.querySelector("updated").textContent : "";
        var link = entry.querySelector("link") ? entry.querySelector("link").getAttribute("href") : "#";

        html += "<tr>";
        html += "<td><div class=\"insider-name\" style=\"font-size:11px;\">" + title + "</div></td>";
        html += "<td class=\"r\"><span class=\"buy-badge\">FORM 4</span></td>";
        html += "<td class=\"r\"><span class=\"date-badge\">" + (updated ? timeAgo(updated) : "—") + "</span></td>";
        html += "<td class=\"r\"><a class=\"sec-link\" href=\"" + link + "\" target=\"_blank\">View &#8599;</a></td>";
        html += "</tr>";
        count++;
      });

      html += "</tbody></table>";
      document.getElementById("insider-container").innerHTML = html;
    })
    .catch(function() {
      document.getElementById("insider-container").innerHTML = "<div class=\"error-state\">Could not load insider filings. <a class=\"sec-link\" href=\"https://www.sec.gov/cgi-bin/browse-edgar?action=getcurrent&type=4&dateb=&owner=include&count=40\" target=\"_blank\">View on SEC.gov &#8599;</a></div>";
    });
}

async function loadInstitutional() {
  var container = document.getElementById("institutional-container");
  try {
    var res = await fetch("https://www.sec.gov/cgi-bin/browse-edgar?action=getcurrent&type=13F-HR&dateb=&owner=include&count=15&search_text=&output=atom");
    var xml = await res.text();
    var parser = new DOMParser();
    var doc = parser.parseFromString(xml, "application/xml");
    var entries = doc.querySelectorAll("entry");

    var html = "";
    var count = 0;
    entries.forEach(function(entry) {
      if (count >= 15) return;
      var title = entry.querySelector("title") ? entry.querySelector("title").textContent : "—";
      var updated = entry.querySelector("updated") ? entry.querySelector("updated").textContent : "";
      var link = entry.querySelector("link") ? entry.querySelector("link").getAttribute("href") : "#";

      var parts = title.split(" - ");
      var fund = parts[0] || title;

      html += "<div class=\"sidebar-item\">";
      html += "<div class=\"sidebar-item-head\">";
      html += "<div class=\"sidebar-company\">" + fund + "</div>";
      html += "<span class=\"sidebar-ticker\">13-F</span>";
      html += "</div>";
      html += "<div class=\"sidebar-meta\">" + (updated ? timeAgo(updated) : "—") + " &bull; <a class=\"sec-link\" href=\"" + link + "\" target=\"_blank\">View filing &#8599;</a></div>";

      count++;

    container.innerHTML = "<div class=\"error-state\">Could not load institutional filings.</div>";
  }
}

function getToday() {
  return new Date().toISOString().split("T")[0];
}
function getDateNDaysAgo(n) {
  var d = new Date();
  d.setDate(d.getDate() - n);
  return d.toISOString().split("T")[0];
}
